Entrada/salida

var Fs = require('fs');

var file = fs.createReadStream('sample.txt', {
    flags: 'r',
    encoding: 'binary',
    start: 10, 
    end: 100,
    buffer: 10
});
file.on('data', function(data) {
    console.log(data);
});
file.on('error', function (excepcion) {
    console.log('error:' + excepcion);
});
file.on('close', function() {
    console.log('chau!');
});

Webserver

var http = require('http');
var i = 0;
http.createServer(function (req, res) {
    console.log('request ' + (i++));
    res.writeHead(200, {'Content-Type': 'text/plain'});
    res.end('Hello World\n');
}).listen(8080, "127.0.0.1");
console.log('Server running at http://127.0.0.1:8080/');

Webserver

var http = require('http');
var i = 0;
http.createServer(function (req, res) {
    var id = i++;
    console.log('request ' + id);
    setTimeout(function() {
        console.log('response ' + id);
        res.writeHead(200, {'Content-Type': 'text/plain'});
        res.end('Hello World: ' + id + ' \n');
    }, 10 * 1000);
}).listen(8080, "127.0.0.1");
console.log('Server running at http://127.0.0.1:8080/');

Cambio de paradigma

// normal
console.log('hello');
pause(5);
console.log('world');

// asíncrono
console.log('hello');
setTimeout(function() {
    console.log('world');
}, 5*1000);

// normal
while(true) {
    console.log('hello');
    console.log('world');
    pause(5);
}

// asíncrono
setInterval(function() {
    console.log('hello');
    console.log('world');
}, 5*1000);

Tradicional

    return DB.find(id).count() == 1;

if (existe(5)) {

Asíncrono

function existe(id, callback) {
    DB.find(id).count(function(total) {
        callback(total == 1);
    });
}

existe(5, function(result) {
    if (result) {
        console.log('OK');
    } else {
        console.log('Error');
});
