\slide{NoSQL, \textquestiondown Qué es?}

\vskip 2in
\begin{center}
{\color{mdwblue}\large\trebucbd NoSQL es un término usado en informática para agrupar una serie de almacenes de datos no relacionales que no proporcionan garantías ACID. Normalmente no tienen esquemas fijos de tablas ni sentencias \textquotedblleft join\textquotedblright.}
\end{center}
\begin{flushright}
{\tt Wikipedia}
\end{flushright}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\slide{NoSQL, \textquestiondown Qué es?}

\vskip 1in
\begin{list1}
    \item Como término nació a principios del 2009
    \item Como concepto, es algo viejo
    \item Conjunto de bases de datos que:
        \begin{list2}
            \item No provee SQL
            \item Los datos no tienen relación
            \item Generalmente escalan de horizontalmente
        \end{list2}
\end{list1}

\slide{\textquotedblleft Problemas\textquotedblright de las bases de datos relacionales}

\vskip 1in
\begin{list1}
    \item SQL
    \begin{list2}
        \item Lenguaje interpretado
        \item Mucha funcionalidad
    \end{list2}
    \item ACID ofrece demasiado
    \item La manera más fácil de escalar es verticalmente
\end{list1}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\slide{\textquotedblleft CAP\textquotedblright y \textquotedblleft BASE\textquotedblright}

\vskip 1in
\begin{list1}
    \item Teorema de CAP, puedes tener dos cosas:
    \begin{list2}
        \item Consistency
        \item Availability
        \item Partition tolerance
    \end{list2}
    \item BASE, el reemplazo de ACID
    \begin{list2}
        \item Basically Available
        \item Soft state
        \item Eventually consistent
    \end{list2}
\end{list1}

\slide{Redis}

\begin{list1}
    \item In-memory Key-value store
    \item Muy rápido, en mi notebook:
    \begin{list2}
        \item 20K inserts en 0.6 segundos
        \item 20K selects en 1 segundo 
    \end{list2}
    \item Soporta operaciones con las claves
    \item Los valores pueden ser:
    \begin{list2}
        \item bytes
        \item Sets
        \item Tuplas
    \end{list2}
    \item Operaciones atómicas (Push, Pop, ranges, etc)
    \item Clientes para varios lenguajes
    \item Replicación a disco
\end{list1}

\slide{CouchDB}

\begin{list1}
    \item Proyecto Apache
    \item JSON (no XML :-)
    \item Interfaz RESTful (+/-)
    \begin{list2}
        \item Estándar (++)
        \item Http es lento (--)
    \end{list2}
    \item MVCC
    \item Safe IO (append only b-tree)
    \item Multi-master
    \item Excelente Interfaz de administración

\end{list1}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% 
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\slide{CouchDB}

\begin{list1}
    \item Javascript incrustado
    \item No soporta indexes (--)
    \item Map/Reduce para realizar queries (+/-)
    \item Soporta guardar archivos
    \item Fácil de escalar (con proxies convencionales)
    \item Distribuido y concurrente por naturaleza (Erlang)
\end{list1}

\slide{MongoDB}

\begin{list1}
    \item <EN>Mongo</EN> != <ES>Mongo</ES>
    \item Orientada a documento
    \item Rápida, escalable, fácil de usar
    \item Soporta índices
    \begin{list2}

        \item Compuestos
        \item Geo-spatial
    \end{list2}
    \item Auto-sharding
    \item Cliente PECL
    \item {\em El \textquotedblleft MySQL\textquotedblright del NoSQL}
\end{list1}
